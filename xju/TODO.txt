- io
  - InputFile
  - OutputFile
  - TCPSocket
      - InByteStream& input()

      - OutByteStream& output()
        - size_t write(void* buffer, size_t bytes, Time until);
  - UDP
      Input& input();
      Output& output();

      - void writeMessage(void* buffer, size_t bytes, Time until);

  - Pipe
      - unique_ptr<InByteStream> input_;

  - SocketPair
  - UnixSocket
  - MessageQueue (linux allows select)
